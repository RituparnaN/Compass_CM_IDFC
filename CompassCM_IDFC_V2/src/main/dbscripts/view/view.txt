CREATE OR REPLACE  VIEW VW_ALERTCODE AS 
SELECT 'ALL' OPTIONNAME, 'ALL' OPTIONVALUE FROM DUAL
UNION ALL
SELECT ALERTCODE OPTIONNAME, ALERTNAME OPTIONVALUE FROM COMAML.TB_ALERTMASTER;

CREATE OR REPLACE VIEW VW_CHANNELTYPE AS SELECT 'ALL' OPTIONNAME, 'ALL' OPTIONVALUE FROM DUAL;

CREATE OR REPLACE VIEW VW_BRANCHCODE AS 
SELECT 'ALL' OPTIONNAME, 'ALL' OPTIONVALUE FROM DUAL
UNION ALL
SELECT BRANCHCODE OPTIONNAME, BRANCHNAME OPTIONVALUE FROM COMAML.TB_BRANCHMASTER;


CREATE OR REPLACE VIEW VW_USERCODE AS SELECT LASTNAME OPTIONNAME,LASTNAME OPTIONVALUE FROM TB_USER;

CREATE OR REPLACE VIEW VW_ACCOUNTNO AS SELECT LASTNAME OPTIONNAME,LASTNAME OPTIONVALUE FROM TB_USER;


CREATE OR REPLACE VIEW VW_CUSTOMERTYPE AS
SELECT 'ALL' OPTIONNAME, 'ALL' OPTIONVALUE FROM DUAL
UNION ALL 
SELECT CUSTOMERTYPE OPTIONNAME,DESCRIPTION OPTIONVALUE FROM COMAML.TB_CUSTOMERTYPEMASTER;

CREATE OR REPLACE VIEW VW_RISKRATING AS SELECT LASTNAME OPTIONNAME,LASTNAME OPTIONVALUE FROM TB_USER;

CREATE OR REPLACE VIEW VW_COUNTRYCODE AS SELECT LASTNAME OPTIONNAME,LASTNAME OPTIONVALUE FROM TB_USER;

CREATE OR REPLACE VIEW VW_REFERENCETYPE AS SELECT LASTNAME OPTIONNAME,LASTNAME OPTIONVALUE FROM TB_USER;

CREATE OR REPLACE VIEW VW_FROMCURRENCY AS SELECT LASTNAME OPTIONNAME,LASTNAME OPTIONVALUE FROM TB_USER;
CREATE OR REPLACE VIEW VW_TOCURRENCY AS SELECT LASTNAME OPTIONNAME,LASTNAME OPTIONVALUE FROM TB_USER;
