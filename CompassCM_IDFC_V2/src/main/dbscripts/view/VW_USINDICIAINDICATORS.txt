	SELECT 'NONE' INDICIANAME, 'NONE' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISGREENCARDHOLDER' INDICIANAME, 'ISGREENCARDHOLDER' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISSUBSTANTIALPRESENCEINUS' INDICIANAME, 'ISSUBSTANTIALPRESENCEINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISSUBSTANTIALPRESENCEINUS' INDICIANAME, 'ISSUBSTANTIALPRESENCEINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISRESIDENCEADDRESSINUS' INDICIANAME, 'ISRESIDENCEADDRESSINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISPOSTALADDRESSINUS' INDICIANAME, 'ISPOSTALADDRESSINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISPLACEOFBIRTHINUS' INDICIANAME, 'ISPLACEOFBIRTHINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISCOUNTRYOFBIRTHINUS' INDICIANAME, 'ISCOUNTRYOFBIRTHINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISCAREOFADDRESSINUS' INDICIANAME, 'ISCAREOFADDRESSINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISHOLDMAILADDRESSINUS' INDICIANAME, 'ISHOLDMAILADDRESSINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISPOBOXNOINUS' INDICIANAME, 'ISPOBOXNOINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISPHONENUMBERINUS' INDICIANAME, 'ISPHONENUMBERINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISSTANDINGINSTRUCTIONTOUSACCT' INDICIANAME, 'ISSTANDINGINSTRUCTIONTOUSACCT' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISTOWNOFINCORPORATIONINUS' INDICIANAME, 'ISTOWNOFINCORPORATIONINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISCOUNTRYOFINCORPORATIONINUS' INDICIANAME, 'ISCOUNTRYOFINCORPORATIONINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISREGISTRATIONADDRESSINUS' INDICIANAME, 'ISREGISTRATIONADDRESSINUS' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISINVOLVEDINACTIVETRADE' INDICIANAME, 'ISINVOLVEDINACTIVETRADE' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISUSPERSONSAS10SHAREHOLDING' INDICIANAME, 'ISUSPERSONSAS10SHAREHOLDING' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISNFFE' INDICIANAME, 'ISNFFE' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISW8W9FORMATTACHED' INDICIANAME, 'ISW8W9FORMATTACHED' DESCRIPTION FROM DUAL
	UNION ALL
	SELECT 'ISRECALCITRANT' INDICIANAME, 'ISRECALCITRANT' DESCRIPTION FROM DUAL
) A;
