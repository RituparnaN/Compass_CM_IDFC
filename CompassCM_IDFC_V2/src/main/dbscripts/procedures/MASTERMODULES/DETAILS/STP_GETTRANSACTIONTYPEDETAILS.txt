CREATE OR REPLACE PROCEDURE STP_GETTRANSACTIONTYPEDETAILS(
i_vcTransTypeCode	 IN VARCHAR2,			i_vcUserCode     IN VARCHAR2, 
i_vcGroupCode        IN VARCHAR2,			i_vcIPAddress    IN VARCHAR2,	   
io_cuResultSet1  	 IN OUT SYS_REFCURSOR,	io_cuResultSet2  IN OUT SYS_REFCURSOR,
io_cuResultSet3  	 IN OUT SYS_REFCURSOR

---EXEC STP_GETTRANSACTIONTYPEDETAILS('100','','','',:RC1,:RC2,:RC3);

)
AS 
vcSql VARCHAR2(32000);

 BEGIN
  
  OPEN io_cuResultSet1 FOR
SELECT 'Transaction Type Details' TABNAMES
  FROM DUAL;
  
  OPEN io_cuResultSet2 FOR
SELECT 'D' TABDISPLAY
  FROM DUAL;

  OPEN io_cuResultSet3 FOR
SELECT TRANSACTIONTYPE, DESCRIPTION, TRANSUBTYPE
  FROM COMAML.TB_TRANSACTIONMASTER WHERE TRANSACTIONTYPE = i_vcTransTypeCode;
  
END STP_GETTRANSACTIONTYPEDETAILS;
/
