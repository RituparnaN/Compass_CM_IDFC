CREATE OR REPLACE PROCEDURE STP_GETBANKDETAILS(
i_vcBankCode	 IN VARCHAR2,			i_vcUserCode     IN VARCHAR2, 
i_vcGroupCode    IN VARCHAR2,			i_vcIPAddress    IN VARCHAR2,	   
io_cuResultSet1  IN OUT SYS_REFCURSOR,	io_cuResultSet2  IN OUT SYS_REFCURSOR,
io_cuResultSet3  IN OUT SYS_REFCURSOR

---EXEC STP_GETBANKDETAILS('INIBKL','','','',:RC1,:RC2,:RC3);

)
AS 
vcSql VARCHAR2(32000);

 BEGIN
  
  OPEN io_cuResultSet1 FOR
SELECT 'Bank Details' TABNAMES
  FROM DUAL;
  
  OPEN io_cuResultSet2 FOR
SELECT 'D' TABDISPLAY
  FROM DUAL;

  OPEN io_cuResultSet3 FOR
SELECT BANKCODE, BANKNAME, BANKTYPE, MDNAME, NOOFBRANCHES, AUTHORISEDFOREXDEALER, UPDATETIMESTAMP, UPDATEDBY,
	   BSRCODE, PRINCIPALOFFICER_EMP_ID, IDALLOCATEDBYFIU, GIINCODE 
  FROM COMAML.TB_BANKMASTER WHERE BANKCODE = i_vcBankCode;

END STP_GETBANKDETAILS;
