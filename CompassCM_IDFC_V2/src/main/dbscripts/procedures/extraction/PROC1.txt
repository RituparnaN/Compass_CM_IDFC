CREATE OR REPLACE PROCEDURE PROC1 (
i_vcFromDate IN VARCHAR2, i_vcToDate IN VARCHAR2, i_vcUserCode IN VARCHAR2)
AS
tsStartDateTime TIMESTAMP(6) := SYSTIMESTAMP;
BEGIN
 INSERT INTO TB_DAILYEXTRACTIONIMPORTLOG VALUES('1','PROC1 Started',SYSTIMESTAMP);
 COMMIT;
 
 DBMS_LOCK.Sleep(45);
 
 INSERT INTO TB_DAILYEXTRACTIONIMPORTLOG VALUES('1','PROC1 Completed',SYSTIMESTAMP);
 COMMIT;
 
 INSERT INTO TB_PROCESSLOG(GROUP_ID, SL_NO, PROC_NAME, STARTTIMESTAMP, ENDTIMESTAMP, LOGTIMESTAMP) VALUES (1, 1, 'PROC1', tsStartDateTime, SYSTIMESTAMP, SYSTIMESTAMP);
 COMMIT;
END PROC1;
